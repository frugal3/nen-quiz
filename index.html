        function showQuestion() {

            const question = questions[currentQuestion];

            document.getElementById('questionText').textContent = `${currentQuestion + 1}. ${question.text}`;

            

            const optionsContainer = document.getElementById('optionsContainer');

            optionsContainer.innerHTML = '';

            

            question.options.forEach((option, index) => {

                const optionDiv = document.createElement('div');

                optionDiv.classList.add('option');

                optionDiv.textContent = option.text;

                optionDiv.onclick = () => selectOption(index);

                optionsContainer.appendChild(optionDiv);

            });

            

            updateProgress();

            updateButtons();

        }


        function selectOption(index) {

            const options = document.querySelectorAll('.option');

            options.forEach(option => option.classList.remove('selected'));

            options[index].classList.add('selected');

            

            answers[currentQuestion] = index;

            document.getElementById('nextBtn').disabled = false;

        }


        function nextQuestion() {

            if (answers[currentQuestion] !== undefined) {

                // Update scores based on selected option

                Object.keys(selectedOption.weights).forEach(trait => {

                    scores[trait] += selectedOption.weights[trait];

                if (currentQuestion >= questions.length) {

                    showResults();

                } else {

                    showQuestion();

                }

            }

        }


        function previousQuestion() {

            if (currentQuestion > 0) {

            }

        }


        function updateProgress() {

            const progress = ((currentQuestion) / questions.length) * 100;

            document.getElementById('progressBar').style.width = `${progress}%`;

        }


        function updateButtons() {

            document.getElementById('prevBtn').disabled = currentQuestion === 0;

            document.getElementById('nextBtn').disabled = answers[currentQuestion] === undefined;

        }


        function calculateNenType() {

            // Determine primary nen type based on scores

            const nenTypes = {

                enhancer: scores.enhancer + (scores.drive * 0.5) + (scores.conscientiousness * 0.3),

                emitter: scores.emitter + (scores.influence * 0.5) + (scores.extroversion * 0.3),

                transmuter: scores.transmuter + (scores.openness * 0.4) + (scores.type3 * 0.3),

                manipulator: scores.manipulator + (scores.clarity * 0.5) + (scores.type1 * 0.3),

                conjurer: scores.conjurer + (scores.conscientiousness * 0.4) + (scores.support * 0.3),

                specialist: scores.specialist + (scores.type5 * 0.5) + (scores.openness * 0.2)

            };


            // Find the highest scoring nen type

            const primaryType = Object.keys(nenTypes).reduce((a, b) => 

                nenTypes[a] > nenTypes[b] ? a : b

            );


            // Find secondary type (for hybrid abilities)

            const sortedTypes = Object.entries(nenTypes).sort((a, b) => b[1] - a[1]);

            const secondaryType = sortedTypes[1][0];


            // Determine personality traits for ability naming

            const personalityTraits = {

                isCreative: scores.openness > 8,

                isAchiever: scores.type3 > 6,

                isAnalytical: scores.type5 > 6,

                isInfluential: scores.influence > 6,

                isMadScientist: scores.openness > 8 && (scores.type3 > 4 || scores.type5 > 4)

            };


            return {

                primary: primaryType,

                secondary: secondaryType,

                traits: personalityTraits,

                scores: nenTypes

            };

        }

                        "You possess a completely unique ability that doesn't fit standard nen categories. Your independent nature has developed something truly one-of-a-kind.",

                    special: result.traits.isAnalytical ? "Pattern Recognition - You can understand and replicate others' nen abilities through observation." : ""

                }

            };


            const primary = abilities[result.primary];

            

            // Add hybrid elements if secondary type is strong

            if (result.scores[result.secondary] > result.scores[result.primary] * 0.8) {

                primary.description += ` You also show strong ${result.secondary} tendencies, making your abilities more versatile than typical ${result.primary}s.`;

            }


            return primary;

        }


        function showResults() {

            const result = calculateNenType();

            const ability = getNenAbility(result);

            

            document.getElementById('quiz').classList.add('hidden');

            document.getElementById('results').classList.remove('hidden');

            

            // Capitalize nen type name

            const nenTypeName = result.primary.charAt(0).toUpperCase() + result.primary.slice(1);

            

            document.getElementById('nenType').textContent = nenTypeName;

            document.getElementById('abilityName').textContent = `"${ability.name}"`;

            

            let description = ability.description;

            if (ability.special) {

                description += `\n\nSpecial Trait: ${ability.special}`;

            }

            

            document.getElementById('description').textContent = description;

        }
